17public class MD5Util {
18
19    /**
20     * Calculates MD5 hash for string. assume string is UTF-8 encoded
21     * @param input string which is going to be encoded into MD5 format
22     * @return  MD5 representation of the input string
23     * @throws NoSuchAlgorithmException
24     */
25    public static String calculateMd5(String input) throws NoSuchAlgorithmException {
26        try {
27            return calculateMd5(input, "UTF-8");
28        } catch (UnsupportedEncodingException e) {
29            return null; // -- impossible --
30        }
31    }
32
33    /**
34     * Calculates MD5 hash for string.
35     * @param input string which is going to be encoded into MD5 format
36     * @param encoding character encoding of the string which is going to be encoded into MD5 format
37     * @return  MD5 representation of the input string
38     * @throws NoSuchAlgorithmException
39     */
40    public static String calculateMd5(String input, String encoding) throws NoSuchAlgorithmException, UnsupportedEncodingException {
41        MessageDigest md = MessageDigest.getInstance("MD5");
42        md.update(input.getBytes(encoding));
43        byte digest[] = md.digest();
44
45        final StringBuilder hexString = new StringBuilder();
46        for (int i = 0; i < digest.length; i++) {
47            int z = 0xFF & digest[i];
48            if (z < 16)
49                hexString.append("0");
50            hexString.append(Integer.toHexString(z));
51        }
52
53        return hexString.toString();
54    }
