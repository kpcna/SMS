8 /**
9  * SendSignedTextMessage.java<br><br>
10 *
11 * An example of using the nexmo sms api to submit a simple text message ...<br><br>
12 * This example uses a {@link com.nexmo.messaging.sdk.NexmoSmsClientSignedRequests} client
13 * So that the REST requests are signed using the secret key.
14 *
15 * Created on 10 April 2012, 17:34
16 *
17 * @author  Paul Cook
18 * @version 1.0
19 */
20public class SendSignedTextMessage {
21
22    public static final String API_KEY = "your-api-key-goes-here";
23    public static final String API_SECRET = "your-api-secret-goes-here";
24
25    public static final String SMS_FROM = "12345";
26    public static final String SMS_TO = "447777111222";
27    public static final String SMS_TEXT = "Hello World!";
28
29    public static void main(String[] args) {
30
31        // Create a client for submitting to Nexmo
32
33        NexmoSmsClient client = null;
34        try {
35            client = new NexmoSmsClientSignedRequests(API_KEY, API_SECRET);
36        } catch (Exception e) {
37            System.err.println("Failed to instanciate a Nexmo Client");
38            e.printStackTrace();
39            throw new RuntimeException("Failed to instanciate a Nexmo Client");
40        }
41
42        // Create a Text SMS Message request object ...
43
44        TextMessage message = new TextMessage(SMS_FROM, SMS_TO, SMS_TEXT);
45
46        // Use the Nexmo client to submit the Text Message ...
47
48        SmsSubmissionResult[] results = null;
49        try {
50            results = client.submitMessage(message);
51        } catch (Exception e) {
52            System.err.println("Failed to communicate with the Nexmo Client");
53            e.printStackTrace();
54            throw new RuntimeException("Failed to communicate with the Nexmo Client");
55        }
56
57        // Evaluate the results of the submission attempt ...
58        System.out.println("... Message submitted in [ " + results.length + " ] parts");
59        for (int i=0;i<results.length;i++) {
60            System.out.println("--------- part [ " + (i + 1) + " ] ------------");
61            System.out.println("Status [ " + results[i].getStatus() + " ] ...");
62            if (results[i].getStatus() == SmsSubmissionResult.STATUS_OK)
63                System.out.println("SUCCESS");
64            else if (results[i].getTemporaryError())
65                System.out.println("TEMPORARY FAILURE - PLEASE RETRY");
66            else
67                System.out.println("SUBMISSION FAILED!");
68            System.out.println("Message-Id [ " + results[i].getMessageId() + " ] ...");
69            System.out.println("Error-Text [ " + results[i].getErrorText() + " ] ...");
70
71            if (results[i].getMessagePrice() != null)
72                System.out.println("Message-Price [ " + results[i].getMessagePrice() + " ] ...");
73            if (results[i].getRemainingBalance() != null)
74                System.out.println("Remaining-Balance [ " + results[i].getRemainingBalance() + " ] ...");
75        }
76    }
77
78}
